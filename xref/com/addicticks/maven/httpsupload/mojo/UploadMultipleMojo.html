30  /**
31   * Uploads multiple file to a remote server using HTTP or HTTPS. This
32   * goal should only be used with servers that accepts <i>multiple</i> file
33   * to be uploaded in a single operation.
34   *
35   * <p>Files are sent using POST method and using
36   * {@code multipart/form-data} encoding (RFC2388). 
37   * 
38   * @author Addicticks
39   */
40  @Mojo(  name = "multiple",
41          requiresProject = true,
42          defaultPhase = LifecyclePhase.DEPLOY)
43  public class UploadMultipleMojo extends UploadAbstractMojo   {
44  
45      
46      
47      
48      /**
49       * Files to upload.
50       * 
51       * <p>
52       * Example:
53       * <pre>
54       *     &lt;uploadFiles&gt;
55       *         &lt;uploadFile&gt;
56       *             &lt;formFieldName&gt;file1&lt;/formFieldName&gt;
57       *             &lt;file&gt;${project.build.directory}/myfile.jar&lt;/file&gt;
58       *         &lt;/uploadFile&gt;
59       *         &lt;uploadFile&gt;
60       *             &lt;formFieldName&gt;file2&lt;/formFieldName&gt;
61       *             &lt;file&gt;c://mydir/whatsup.dat&lt;/file&gt;
62       *             &lt;hintFilename&gt;whatsup.xls&lt;/hintFilename&gt;
63       *         &lt;/uploadFile&gt;
64       *     &lt;/uploadFiles&gt;
65       * </pre>
66       * 
67       * <p>
68       * There must be at least one <code>uploadFile</code> element.
69       * 
70       * <p>
71       * The elements of an <code>uploadFile</code> are as follows:
72       * <ul>
73       *    <li><code>formFieldName</code>. The form field name to POST the file to. This value depends entirely on
74       *          the server you're uploading to. Typically such a field is named "file" 
75       *          or if multiple files are allowed: "file1", "file2", etc.
76       *          The value must be unique among the files to be uploaded.
77       *          <br>Type: <code>String</code>, Required: <code>Yes</code>
78       *    </li><br>
79       *    <li><code>file</code>. File to upload. 
80       *          <br>Type: <code>java.io.File</code>, Required: <code>Yes</code>
81       *    </li><br>
82       *    <li><code>mimeType</code>. This is the http <code>Content-Type</code> 
83       *          used when the file is uploaded. If not given explicitly here it 
84       *          will be derived from the file name extension using Java's 
85       *          {@link java.net.URLConnection#guessContentTypeFromName(java.lang.String)}
86       *          method.
87       *          <br>Type: <code>String</code>, Required: <code>No</code>
88       *    </li><br>
89       *    <li><code>hintFilename</code>. Hint given to the server as to what the 
90       *          server should name the file. If this value is not set explicitly 
91       *          it will be derived from the filename of the {@code file} element.
92       *          <br>Type: <code>String</code>, Required: <code>No</code>
93       *    </li>
94       * </ul>
95       * 
96       * 
97       * <br>
98       * <br>
99       **/ 
100     @Parameter(required=true)
101     private List<UploadFile> uploadFiles;
102     
103     
104 
105     
106     
107     
108     @Override
109     public void execute() throws MojoExecutionException, MojoFailureException {
110         validateUploadFiles();
111         
112         HttpsFileUploaderConfig config = getConfig();
113 
114       
115         Map<String, UploadItem> newMap = new HashMap<>();
116         if (uploadFiles != null) {
117             for (UploadFile e : uploadFiles) {
118                 
119                 UploadItem f;
120                 boolean hintFileNameSpecified = true;
121                 boolean mimeTypeSpecified = true;
122                 if (e.getHintFilename() == null) {
123                    hintFileNameSpecified = false;
124                 }
125                 if (e.getMimeType() == null) {
126                     mimeTypeSpecified = false;
127                 }
128                 
129                 if (mimeTypeSpecified && hintFileNameSpecified) {
130                     f = new UploadItemFile(
131                             e.getFile(),
132                             e.getHintFilename(),
133                             e.getMimeType());
134                 } else if (!mimeTypeSpecified && hintFileNameSpecified) {
135                     f = new UploadItemFile(
136                             e.getFile(), 
137                             e.getHintFilename());
138                 } else {
139                     f = new UploadItemFile(
140                             e.getFile());
141                 }
142                 if (newMap.put(e.getFormFieldName(), f) != null) {
143                     throw new MojoExecutionException("The value for formFieldName must be unique among uploadFiles. (value = \"" + e.getFormFieldName() + "\")");
144                 }
145             }
146         }
147         
148         upload(config, newMap, extraFields);
149     }
150 
151     
152     private void validateUploadFiles() throws MojoExecutionException {
153         for (UploadFile e : uploadFiles) {
154             if (e.getFormFieldName() == null || e.getFormFieldName().isEmpty()) {
155                 throw new MojoExecutionException("'formFieldName' is required on 'uploadFile'");
156             }
157             if (e.getFile() == null) {
158                 throw new MojoExecutionException("'file' is required on 'uploadFile'");
159             }
160         }
161         
162     }
